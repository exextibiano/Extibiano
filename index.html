-- Extensões úteis create extension if not exists pgcrypto; -- ===================================================== -- TABELA DE BOSSES -- ===================================================== create table if not exists public.bosses ( id uuid primary key default gen_random_uuid(), slug text unique not null, name text not null, category text not null default 'daily', difficulty text not null default 'medium', party_type text not null default 'solo', -- solo | duo | team level_min integer, level_max integer, cooldown_hours integer, bosstiary_total_kills integer not null default 25, bosstiary_points integer not null default 0, is_active boolean not null default true, created_at timestamptz not null default now() ); create index if not exists idx_bosses_category on public.bosses(category); create index if not exists idx_bosses_level_min on public.bosses(level_min); create index if not exists idx_bosses_level_max on public.bosses(level_max); -- ===================================================== -- PERFIL DO PERSONAGEM -- Cada usuário pode ter vários chars. -- ===================================================== create table if not exists public.characters ( id uuid primary key default gen_random_uuid(), user_id uuid not null references auth.users(id) on delete cascade, name text not null, world text, vocation text, level integer, is_public boolean not null default false, created_at timestamptz not null default now(), updated_at timestamptz not null default now() ); create index if not exists idx_characters_user_id on public.characters(user_id); create unique index if not exists idx_characters_user_name_unique on public.characters(user_id, name); -- ===================================================== -- ANALYSES DE BOSSES -- Cada registro é uma kill/análise colada no site. -- ===================================================== create table if not exists public.boss_analyses ( id uuid primary key default gen_random_uuid(), user_id uuid not null references auth.users(id) on delete cascade, character_id uuid references public.characters(id) on delete set null, boss_id uuid not null references public.bosses(id) on delete cascade, occurred_at timestamptz not null default now(), analyzer_text_raw text not null, analyzer_text_edited text, duration_seconds integer, damage_done numeric(14,2), healing_done numeric(14,2), loot_value numeric(14,2) not null default 0, supply_value numeric(14,2) not null default 0, profit_value numeric(14,2) generated always as (loot_value - supply_value) stored, xp_gained numeric(14,2) default 0, was_edited boolean not null default false, edit_count integer not null default 0, confidence_score numeric(5,2) not null default 1.00, is_outlier boolean not null default false, is_flagged boolean not null default false, include_in_global_stats boolean not null default true, privacy_mode text not null default 'anonymous', -- private | anonymous | public notes text, created_at timestamptz not null default now(), updated_at timestamptz not null default now() ); create index if not exists idx_boss_analyses_user_id on public.boss_analyses(user_id); create index if not exists idx_boss_analyses_boss_id on public.boss_analyses(boss_id); create index if not exists idx_boss_analyses_character_id on public.boss_analyses(character_id); create index if not exists idx_boss_analyses_occurred_at on public.boss_analyses(occurred_at desc); create index if not exists idx_boss_analyses_include_global on public.boss_analyses(include_in_global_stats); -- ===================================================== -- CAMPOS EDITADOS MANUALMENTE -- Guarda exatamente o que o usuário mexeu. -- ===================================================== create table if not exists public.boss_analysis_edits ( id uuid primary key default gen_random_uuid(), analysis_id uuid not null references public.boss_analyses(id) on delete cascade, field_name text not null, old_value text, new_value text, created_at timestamptz not null default now() ); create index if not exists idx_boss_analysis_edits_analysis_id on public.boss_analysis_edits(analysis_id); -- ===================================================== -- PROGRESSO DE BOSSTIARY POR PERSONAGEM -- ===================================================== create table if not exists public.character_bosstiary_progress ( id uuid primary key default gen_random_uuid(), user_id uuid not null references auth.users(id) on delete cascade, character_id uuid not null references public.characters(id) on delete cascade, boss_id uuid not null references public.bosses(id) on delete cascade, kills_done integer not null default 0, manual_adjustment integer not null default 0, tracked_from_analyses integer not null default 0, total_kills integer generated always as (kills_done + manual_adjustment + tracked_from_analyses) stored, completed boolean not null default false, remaining_to_complete integer not null default 0, points_earned integer not null default 0, created_at timestamptz not null default now(), updated_at timestamptz not null default now(), unique(character_id, boss_id) ); create index if not exists idx_character_bosstiary_character_id on public.character_bosstiary_progress(character_id); create index if not exists idx_character_bosstiary_boss_id on public.character_bosstiary_progress(boss_id); -- ===================================================== -- RANKING CACHE -- Facilita o carregamento rápido. -- ===================================================== create table if not exists public.boss_ranking_cache ( id uuid primary key default gen_random_uuid(), boss_id uuid not null references public.bosses(id) on delete cascade, level_range_label text not null default 'all', sample_size integer not null default 0, unique_players integer not null default 0, median_profit numeric(14,2) not null default 0, avg_profit_trimmed numeric(14,2) not null default 0, median_loot numeric(14,2) not null default 0, median_supply numeric(14,2) not null default 0, median_xp numeric(14,2) not null default 0, risk_rate numeric(8,4) not null default 0, confidence_label text not null default 'low', updated_at timestamptz not null default now(), unique(boss_id, level_range_label) ); -- ===================================================== -- TRIGGERS DE UPDATED_AT -- ===================================================== create or replace function public.set_updated_at() returns trigger language plpgsql as $$ begin new.updated_at = now(); return new; end; $$; drop trigger if exists trg_characters_updated_at on public.characters; create trigger trg_characters_updated_at before update on public.characters for each row execute function public.set_updated_at(); drop trigger if exists trg_boss_analyses_updated_at on public.boss_analyses; create trigger trg_boss_analyses_updated_at before update on public.boss_analyses for each row execute function public.set_updated_at(); drop trigger if exists trg_character_bosstiary_progress_updated_at on public.character_bosstiary_progress; create trigger trg_character_bosstiary_progress_updated_at before update on public.character_bosstiary_progress for each row execute function public.set_updated_at(); -- ===================================================== -- FUNÇÃO: recalcula remaining_to_complete e completed -- ===================================================== create or replace function public.recalc_bosstiary_progress(p_character_id uuid, p_boss_id uuid) returns void language plpgsql as $$ declare v_total integer; v_required integer; v_points integer; begin select total_kills into v_total from public.character_bosstiary_progress where character_id = p_character_id and boss_id = p_boss_id; select bosstiary_total_kills, bosstiary_points into v_required, v_points from public.bosses where id = p_boss_id; update public.character_bosstiary_progress set remaining_to_complete = greatest(v_required - coalesce(v_total, 0), 0), completed = (coalesce(v_total, 0) >= v_required), points_earned = case when coalesce(v_total, 0) >= v_required then v_points else 0 end where character_id = p_character_id and boss_id = p_boss_id; end; $$; -- ===================================================== -- FUNÇÃO: soma +1 no tracked_from_analyses ao salvar análise -- ===================================================== create or replace function public.increment_bosstiary_from_analysis() returns trigger language plpgsql as $$ begin if new.character_id is not null then insert into public.character_bosstiary_progress ( user_id, character_id, boss_id, tracked_from_analyses, remaining_to_complete ) values ( new.user_id, new.character_id, new.boss_id, 1, 0 ) on conflict (character_id, boss_id) do update set tracked_from_analyses = public.character_bosstiary_progress.tracked_from_analyses + 1; perform public.recalc_bosstiary_progress(new.character_id, new.boss_id); end if; return new; end; $$; drop trigger if exists trg_increment_bosstiary_from_analysis on public.boss_analyses; create trigger trg_increment_bosstiary_from_analysis after insert on public.boss_analyses for each row execute function public.increment_bosstiary_from_analysis();
